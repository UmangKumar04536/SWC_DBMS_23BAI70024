SELECT EXTRACT (MONTH FROM SUBMIT_DATE) AS MTH, PRODUCT_ID AS PRODUCT,
ROUND(AVG(STARS),2) AS AVG_STARS
FROM REVIEWS 
GROUP BY EXTRACT(MONTH FROM SUBMIT_DATE),PRODUCT_ID
ORDER BY MTH,PRODUCT

WITH CTE AS(
  SELECT EXTRACT(YEAR FROM TRANSACTION_DATE) AS YEAR,
  PRODUCT_ID,SUM(SPEND) AS CURR_YEAR_SPEND
  FROM USER_TRANSACTIONS
  GROUP BY 1,2
)
SELECT YEAR,PRODUCT_ID,CURR_YEAR_SPEND,PREV_YEAR_SPEND,
ROUND(100.0 *(CURR_YEAR_SPEND - PREV_YEAR_SPEND)/ PREV_YEAR_SPEND,2 ) AS YOY_RATE
FROM(
  SELECT YEAR, PRODUCT_ID,CURR_YEAR_SPEND,
  LAG(CURR_YEAR_SPEND ) OVER (PARTITION BY PRODUCT_ID  ORDER BY YEAR) AS PREV_YEAR_SPEND
  FROM CTE
)T
ORDER BY PRODUCT_ID,YEAR
